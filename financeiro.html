Tabela de despesas nao encontrada no Supabase
Execute este SQL no Supabase Dashboard → SQL Editor para criar a tabela de contas a pagar:
CREATE TABLE contas_pagar (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  salao_id UUID REFERENCES saloes(id) ON DELETE CASCADE,
  descricao TEXT NOT NULL,
  valor NUMERIC(10,2) NOT NULL,
  data_vencimento DATE NOT NULL,
  data_pagamento DATE,
  pago BOOLEAN DEFAULT FALSE,
  categoria TEXT DEFAULT 'outros',
  forma_pagamento TEXT,
  observacoes TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
ALTER TABLE contas_pagar ENABLE ROW LEVEL SECURITY;
CREATE POLICY "salao_own" ON contas_pagar
  FOR ALL USING (
    salao_id = (SELECT id FROM saloes WHERE usuario_id = auth.uid() LIMIT 1)
  );
